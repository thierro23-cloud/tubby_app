<script>
document.getElementById("tipo_via").addEventListener("change", function() {

    const idTipo = this.value;

    fetch(`/contenedores_validacion/calles_por_tipo/${idTipo}`)
        .then(res => res.json())
        .then(data => {

            const selectCalle = document.getElementById("calle");
            selectCalle.innerHTML = '<option value="">-- Seleccionar --</option>';

            data.forEach(calle => {
                const option = document.createElement("option");
                option.value = calle.idtbl_calles;
                option.textContent = calle.calles;
                selectCalle.appendChild(option);
            });

        });

});
</script>
