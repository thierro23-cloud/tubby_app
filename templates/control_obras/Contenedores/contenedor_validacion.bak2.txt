     7. JS: CALLES DEPENDIENTES DEL TIPO DE VÍA

<script>
document.addEventListener('DOMContentLoaded', function () {
    const tipoVia = document.getElementById("tipo_via");
    const calleSelect = document.getElementById("calle");

    tipoVia.addEventListener("change", function() {
        const idTipo = this.value;

        if (!idTipo) {
            calleSelect.innerHTML = '<option value="">Selecciona calle…</option>';
            return;
        }

        fetch(`/contenedores_validacion/calles_por_tipo/${idTipo}`)
            .then(r => r.json())
            .then(data => {
                calleSelect.innerHTML = '<option value="">Selecciona calle…</option>';
                data.forEach(x => {
                    const o = document.createElement("option");
                    o.value = x.idtbl_calles;
                    o.textContent = x.calles;
                    calleSelect.appendChild(o);
                });
            });
    });
});
</script>
